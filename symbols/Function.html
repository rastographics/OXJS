(Function) around()

Provides access to the overridden base function. If there is no base function to override, the first argument to the function will be an empty function that returns nothing.

The first argument passed in is the base function, bound to the correct scope (so you don't need to call apply/call).


Returns:
(Function) The reciever.

  var foo = OX.Base.extend({
    bar: function (junk) {
      return 'bar' + junk;
    }
  });
  var fooBar = foo.extend({
    bar: function ($super, junk) {
      return 'foo' + $super(junk);
    }.around()
  });

  foo.bar('bell');
  // -> 'barbell'
  fooBar.bar('n');
  // -> 'foobarn'

(Function) inferior()

Marks the function as inferior. If a key exists on the mixin, and the new mixin is marked as inferior, it will ignore the new function, relying on the old function for its implementation.
Returns:
(Function) The reciever.

  var foo = OX.Base.extend({
    bar: function () {
      return "superior";
    },
  }, {
    bar: function () {
      return "inferior";
    }.inferior()
  });

  foo.bar();
  // -> "superior"
