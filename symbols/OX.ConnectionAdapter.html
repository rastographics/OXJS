Overview

OX Connection Adapter abstract object. An instance of this object MUST be supplied to the OX.Connection instance. This object is to be defined by consumers of the API as an adapter to the XMPP connection library that is being used. The example below shows a sample implementation for the OX.ConnectionAdapter using the JSJaC XMPP library.

Extends:
OX.Base

  var conn = new JSJaCConnection({ httpbase: '/http-bind/' });
  var adapter = OX.ConnectionAdapter.extend({
    jid: function () { return conn.fulljid; },

    registerHandler: function (event, handler) {
      return conn.registerHandler(event, handler);
    },

    unregisterHandler: function (event, handler) {
      return conn.unregisterHandler(event, handler);
    },

    send: function (xml, cb, args) {
      return conn._sendRaw(xml, cb, args);
    }
  });

  var ox = OX.Connection.extend({ connectionAdapter: adapter });
