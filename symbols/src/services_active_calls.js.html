1 /**
  2  * @namespace
  3  * <p>Namespace for active-calls related services.</p>
  4  *
  5  * <p>Active Calls pubsub allows for subscribing to
  6  * evented data regarding the current calls of a SIP address.
  7  * Data returned in the events may be used for call control
  8  * purposes in the Active Calls command component or
  9  * for other application specific usage.</p>
 10  *
 11  * <p>The Active-Calls component is responsible for
 12  * all third party call control (3PCC) requests for the XMPP API.
 13  * Each command node of the Active Calls component performs
 14  * a specific 3PCC function, see below for more information.</p>
 15  *
 16  * @extends OX.Base
 17  * @extends OX.Mixin.Subscribable
 18  * @requires A 'connectionAdapter' property inherited from an {@link OX.Connection}.
 19  * @requires Authentication to interact with PubSub via {@link OX.Service.Auth}.
 20  * @see <a href="http://wiki.onsip.com/docs/Active-Calls_Pubsub">ActiveCalls PubSub</a>
 21  * @see <a href="http://wiki.onsip.com/docs/Active-Calls_Component">ActiveCalls Component</a>
 22  */
 23 OX.Service.ActiveCalls = OX.Base.extend(OX.Mixin.Subscribable, /** @lends OX.Service.ActiveCalls */ {
 24 
 25   /**
 26    * URI for this PubSub service.
 27    */
 28   pubSubURI: OX.Settings.URIs.pubSub.activeCalls,

 30   /**
 31    * @namespace
 32    * Active Call Item.
 33    * @name OX.Service.ActiveCalls.Item
 34    * @extends OX.Item
 35    * @extends OX.Mixin.CallDialog
 36    */
 37   Item: OX.Item.extend(OX.Mixin.CallDialog,/** @lends OX.Service.ActiveCalls.Item# */{
 38     /** The current dialog state. */
 39     dialogState: null,
 40 
 41     /** The call ID of this call. */
 42     callID: null,
 43 
 44     /** The URI of the call originator. */
 45     fromURI: null,
 46 
 47     /** The URI of the call terminator. */
 48     toURI: null,
 49 
 50     /** The Address of Record for the User Agent Server. */
 51     toAOR: null,
 52 
 53     /** The tag for the originating leg of the call. */
 54     fromTag: null,
 55 
 56     /** The tag for the terminating leg of the call. */
 57     toTag: null,
 58 
 59     /** Caller ID information for the caller */
 60     fromDisplay: null,
 61 
 62     /** Caller ID information for the callee */
 63     toDisplay: null,
 64 
 65     /** The branch tag for this pre-dialog event. */
 66     branch: null,
 67 
 68     /** The tag inserted into the call-setup-id field */
 69     callSetupID: null,
 70 
 71     /**
 72      * <p>When calls are created, the call's origination will be from
 73      * Junction Network's Call-Setup Server.</p>
 74      *
 75      * <p>This lets you know if the call is from the call-setup
 76      * server or not</p>
 77      * @returns {Boolean} Whether the call was from the callSetupID.
 78      */
 79     isFromCallSetup: function () {
 80       return !!this.callSetupID;
 81     },
 82 
 83     /**
 84      * @returns {Boolean} Whether the call was created or not.
 85      */
 86     isCreated: function () {
 87       return this.dialogState === 'created';
 88     },
 89 
 90     /**
 91      * @returns {Boolean} Whether the call was requested.
 92      */
 93     isRequested: function () {
 94       return this.dialogState === 'requested';
 95     },
 96 
 97     /**
 98      * @returns {Boolean} Whether the call was confirmed.
 99      */
100     isConfirmed: function () {
101       return this.dialogState === 'confirmed';
102     },
103 
104     /**
105      * The XMPP-API requires one of two commands to be called to end
106      * a call based on whether or not the call has been answered (confirmed).
107      * This is a convenience funtion to make the correct API call based
108      * upon the dialog state of the current this object.
109      *
110      * @param {Object} [callbacks] An object supplying functions for 'onSuccess' and 'onError'.
111      *   @param {Function} [callbacks.onSuccess] The success callback.
112      *   @param {Function} [callbacks.onError] The error callback.
113      *     @param {OX.PacketAdapter} [callbacks.onError.packet] The packet received.
114      * @returns {void}
115      */
116     hangup: function (callbacks) {
117       return this.isConfirmed() ? this.terminate(callbacks) : this.cancel(callbacks);
118     }

122   /** @private
123    * Returns an {@link OX.Service.ActiveCalls.Item} from an XML Document.
124    * This method should be called once for each item to be constructed.
125    * If a DOM Element contains more than one item node, only the first
126    * item node will be returned as an {@link OX.Service.ActiveCalls.Item}
127    *
128    * @param {Element|Node} element The DOM Element or Node to parse into a {@link OX.Service.ActiveCalls.Item}
129    * @returns {OX.Service.ActiveCalls.Item} item
130    */
131   itemFromElement: function (element) {
132     if (!element) {
133       return undefined;
134     }
135 
136     var activeCallNode = element.getElementsByTagName('active-call'),
137         attrs          = {connection: this.connection};
138 
139     if (!activeCallNode || !activeCallNode[0]) {
140       return undefined;
141     }
142 
143     var childNodes = activeCallNode[0].childNodes;
144     function getFirstNodeValue(node) {
145       var child = node.firstChild;
146       if (child && child.nodeValue === null && child.firstChild) {
147         return arguments.callee(child);
148       } else if (child && child.nodeValue) {
149         return child.nodeValue;
150       }
151       return undefined;
152     }
153 
154     for (var i = 0, len = childNodes.length; i < len; i++) {
155       var node = childNodes[i];
156 
157       if (!node.nodeName) {
158         continue;
159       }
160 
161       switch (node.nodeName.toLowerCase()) {
162       case 'dialog-state':
163         attrs.dialogState = node.firstChild.nodeValue;
164         break;
165       case 'call-id':
166         attrs.callID = node.firstChild.nodeValue;
167         break;
168       case 'to-aor':
169         attrs.toAOR = node.firstChild && node.firstChild.nodeValue;
170         break;
171       case 'from-uri':
172         attrs.fromURI = node.firstChild.nodeValue;
173         break;
174       case 'to-uri':
175         attrs.toURI = node.firstChild.nodeValue;
176         break;
177       case 'from-tag':
178         attrs.fromTag = node.firstChild.nodeValue;
179         break;
180       case 'to-tag':
181         attrs.toTag = node.firstChild && node.firstChild.nodeValue;
182         break;
183       case 'branch':
184         attrs.branch = node.firstChild && node.firstChild.nodeValue;
185         break;
186       case 'call-setup-id':
187         attrs.callSetupID = node.firstChild && node.firstChild.nodeValue;
188         break;
189       case 'from-display':
190         attrs.fromDisplay = node.firstChild && node.firstChild.nodeValue;
191         break;
192       case 'to-display':
193         attrs.toDisplay = node.firstChild && node.firstChild.nodeValue;
194         break;
195       }
196     }
197 
198     return this.Item.extend(attrs);
199   },

201   /**
202    * <p>Create a new call. Note that an 'onSuccess' does <b>NOT</b> mean that call you requested
203    * has started, rather it means the XMPP API has handled your IQ (Info Query) stanza and sent the
204    * SIP message to kick off the call.</p>
205    *
206    * <p>You need to monitor ActiveCalls in PubSub to be notified when a call is actually created.</p>
207    *
208    * <p>The 'callSetupID' <b>MUST</b> conform to the following regular expression for it to be valid:
209    * <pre>/[a-z0-9]{8}/</pre></p>
210    *
211    * <p>In addition, the 'to' and 'from' parameters should be in the 'sip' schema
212    * to make SIP calls (which means calls should be prefixed by 'sip:').</p>
213    *
214    * @param {String} to the SIP address to terminate the call at
215    * @param {String} from the SIP address to originate the call from
216    * @param {String} callSetupID the end to end call tracking code to be used for a call setup
217    * @param {Object} [callbacks] Callbacks with 'onSuccess' and 'onError'
218    *   @param {Function} [callbacks.onSuccess] The success callback
219    *     @param {OX.PacketAdapter} [callbacks.onSuccess.packet] The packet recieved.
220    *   @param {Function} [callbacks.onError] The error callback
221    *     @param {OX.PacketAdapter} [callbacks.onError.packet] The packet recieved.
222    * @returns {void}
223    * @example
224    *   // Assuming the organization is apple.onsip.com that you Authenticated for,
225    *   // and the users existed, these should be valid.
226    *
227    *   // Have Paul call John.
228    *   ox.ActiveCalls.create('sip:john@apple.onsip.com', 'sip:paul@apple.onsip.com', 'drrobert');
229    *
230    *   // Have George call 1-800-801-3381.
231    *   ox.ActiveCalls.create('sip:18008013381@apple.onsip.com', 'sip:george@apple.onsip.com', 'longlong');
232    */
233   create: function (to, from, callSetupID, cb) {
234     var uri   = OX.Settings.URIs.command.createCall,
235         xData = OX.XML.XMPP.XDataForm.create({type: 'submit'}),
236         cmd   = OX.XML.XMPP.Command.create({node: uri.queryParam('node')}, xData),
237         iq    = OX.XML.XMPP.IQ.create({to: uri.path, type: 'set'}, cmd);
238 
239     xData.addField('to', to);
240     xData.addField('from', from);
241     xData.addField('call-setup-id', callSetupID);
242 
243     cb = cb || {};
244 
245     this.connection.send(iq.toString(), function (packet) {
246       if (!packet) {
247         return;
248       }
249 
250       if (packet.getType() === 'error' &&
251           cb.onError && cb.onError.constructor === Function) {
252         cb.onError(packet);
253       } else if (cb.onSuccess && cb.onSuccess.constructor === Function) {
254         cb.onSuccess(packet);
255       }
256     }, []);
257   }
