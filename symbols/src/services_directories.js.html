1 /**
  2  * @namespace
  3  * <p>Namespace for directory related services.</p>
  4  * <p>The Directories Service provides subscribed users to
  5  * recieve updates regarding their organization's directory.</p>
  6  * <p>There are two node levels that a user can subscribe to:</p>
  7  *  <ul>
  8  *    <li>/me/jid</li>
  9  *    <li>/sip-domain</li>
 10  *  </ul>
 11  * <p>Different results <i>may</i> be generated depending upon which node is subscribed to.</p>
 12  * <p>Please note that subscribing to '/' will redirect your subscription request to '/me/jid'.</p>
 13  * @extends OX.Base
 14  * @extends OX.Mixin.Subscribable
 15  * @requires A 'connectionAdapter' property inherited from an {@link OX.Connection}.
 16  * @see <a href="http://wiki.onsip.com/docs/Directories_Pubsub">Directories PubSub</a>
 17  */
 18 OX.Service.Directories = OX.Base.extend(OX.Mixin.Subscribable, /** @lends OX.Service.Directories */{
 19 
 20   /**
 21    * URI for the PubSub directories service
 22    */
 23   pubSubURI: OX.Settings.URIs.pubSub.directories,
 24 
 25   /**
 26    * @class
 27    * A directory item.
 28    * @extends OX.Item
 29    */
 30   Item: OX.Item.extend(/** @lends OX.Service.Directories.Item# */{
 31     /**
 32      * The SIP URI of the entry.
 33      * @type String
 34      */
 35     sipURI: null,
 36 
 37     /**
 38      * The related SIP address.
 39      * @type OX.URI
 40      */
 41     related: null,
 42 
 43     /**
 44      * The ID of the item.
 45      * @returns {String} The item attribute of the URI query.
 46      */
 47     id: function () {
 48       return this.uri.queryParam('item');
 49     }
 50   })
 51 
 52 });
 53 
 54 OX.Service.Directories.mixin(/** @lends OX.Service.Directories */{
 55   /**
 56    * @class
 57    * An alias for a contact.
 58    * @extends OX.Service.Directories.Item
 59    */
 60   AliasItem: OX.Service.Directories.Item.extend(/** @lends OX.Service.Directories.AliasItem# */{
 61     /**
 62      * The XMPP URI of the alias.
 63      * @type OX.URI
 64      */
 65     xmppURI: null
 66   }),
 67 
 68   /**
 69    * @class
 70    * A contact.
 71    * @extends OX.Service.Directories.Item
 72    */
 73   EntityItem: OX.Service.Directories.Item.extend(/** @lends OX.Service.Directories.EntityItem# */{
 74     /**
 75      * The name of the entity.
 76      * @type OX.URI
 77      */
 78     name: null
 79   }),
 80 
 81   /** @private
 82    * Returns the appropriate {@link OX.Service.Directories.Item}
 83    * ({@link OX.Service.Directories.AliasItem} or {@link OX.Service.Directories.EntityItem})
 84    * from an XML Document.
 85    * This method should be called once for each item to be constructed.
 86    * If a DOMElement contains more than one item node, only the first
 87    * item node will be returned as an {@link OX.Service.Directories.Item}
 88    *
 89    * @param {Element|Node} element The DOM Element or Node to parse into a {@link OX.Service.Directories.Item}
 90    * @returns {OX.Service.ActiveCalls.Item} The item.
 91    */
 92   itemFromElement: function (element) {
 93     if (!element) {
 94       return undefined;
 95     }
 96 
 97     var aliasNode =  element.getElementsByTagName('alias'),
 98         entityNode = element.getElementsByTagName('entity'),
 99         node       = aliasNode[0] || entityNode[0],
100         attrs      = { connection: this.connection };
101 
102     if (!node) {
103       return undefined;
104     }
105 
106     var childNodes = node.childNodes;
107 
108     for (var i = 0, len = childNodes.length; i < len; i++) {
109       var childNode = childNodes[i],
110           childNodeName = childNode.nodeName,
111           value = (childNode && childNode.firstChild && childNode.firstChild.nodeValue) || undefined;
112 
113       if (!childNode) {
114         continue;
115       }
116 
117       switch (childNodeName.toLowerCase()) {
118       case 'sip-uri':
119         attrs.sipURI = value;
120         break;
121       case 'xmpp-uri':
122         attrs.xmppURI = OX.URI.parse(value);
123         break;
124       case 'link':
125         if (childNode.getAttribute('rel') === 'related') {
126           attrs.related = OX.URI.parse(childNode.getAttribute('href'));
127         } else {
128           OX.Log.warn('Links with rel="' + childNode.getAttribute('rel') + '" are not supported.');
129         }
130         break;
131       case 'name':
132         attrs.name = value;
133         break;
134       }
135     }
136 
137     return aliasNode[0] ? this.AliasItem.extend(attrs) : this.EntityItem.extend(attrs);
138   }
139 });
