1 /**
  2  * @namespace
  3  * URI namespace, as defined in <a href="http://www.ietf.org/rfc/rfc2396.txt">RFC 2396</a>.
  4  * @extends OX.Base
  5  */
  6 OX.URI = OX.Base.extend(/** @lends OX.URI */{
  7 
  8   /**
  9    * Parse a string into an OX.URI.Base object.
 10    *
 11    * @param {String} uriString the URI to parse
 12    * @returns {OX.URI.Base} A new OX.URI.Base object
 13    *
 14    * @example
 15    *   var uri = OX.URI.parse('xmpp:lisa@example.com');
 16    *   // -> { path: "lisa@example.com", scheme: "xmpp" }
 17    */
 18   parse: function (uriString) {
 19     var scheme, authority, path, query, fragment;
 20 
 21     // Scan for : to find scheme                    - required
 22     // Scan between // and / to find authority      - optional
 23     // Scan from end of authority to ? to find path - required
 24     // Scan from ? to # to find query               - optional
 25     // Scan from # to EOL to find fragment          - optional
 26     var parts = uriString.match(/^([^:]*:)(\/\/[^\/]*\/)?([^?]*)(\?[^#]*)?(#.*)?/);
 27     if (parts[1]) {
 28       scheme = parts[1].substr(0, parts[1].length - 1);
 29     }
 30     if (parts[2]) {
 31       authority = parts[2].substr(2, parts[2].length - 2).substr(0, parts[2].length - 3);
 32     }
 33     if (parts[3]) {
 34       path = parts[3];
 35     }
 36     if (parts[4]) {
 37       query = parts[4].substr(1, parts[4].length - 1);
 38     }
 39     if (parts[5]) {
 40       fragment = parts[5].substr(1, parts[5].length - 1);
 41     }
 42 
 43     return OX.URI.Base.extend({scheme: scheme, authority: authority,
 44                                path: path, query: query, fragment: fragment});
 45   },
 46 
 47   /**
 48    * Convert an object into an OX.URI.Base object.
 49    *
 50    * @param {Object} object an object with these members: scheme, path, authority, query, fragment
 51    * @returns {OX.URI.Base} A new URI object
 52    *
 53    * @example
 54    * var uri = OX.URI.fromObject({scheme: 'xmpp', path: 'lisa@example.com'});
 55    */
 56   fromObject: function (object) {
 57     return OX.URI.Base.extend(object);
 58   }
 59 });
 60 
 61 /**
 62  * @namespace
 63  * Traits object for URI.
 64  * @extends OX.Base
 65  */
 66 OX.URI.Base = OX.Base.extend(/** @lends OX.URI.Base# */{
 67 
 68   /**
 69    * The URI scheme.
 70    *
 71    * @default "xmpp"
 72    */
 73   scheme: 'xmpp',
 74 
 75   /**
 76    * The URI authority section.
 77    */
 78   authority: null,
 79 
 80   /**
 81    * The URI path.
 82    */
 83   path: null,
 84 
 85   /**
 86    * The URI query parameters.
 87    */
 88   query: null,
 89 
 90   /**
 91    * The URI fragment identifier.
 92    */
 93   fragment: null,
 94 
 95   /**
 96    * Return the action, if any, in the query parameters.
 97    *
 98    * @example
 99    *   var uri = OX.URI.parse('xmpp:lisa@example.com?message');
100    *   uri.action();
101    * @returns {String} The action of this query, or undefined if none found.
102    */
103   action: function () {
104     if (!this.query) {
105       return undefined;
106     }
107 
108     var parts = this.query.split(';');
109     if (parts[0] === '') {
110       return undefined;
111     } else {
112       return parts[0];
113     }
114   },
115 
116   /**
117    * Return the value, if any, for a parameter in the query
118    * parameters.
119    *
120    * @example
121    *   var uri = OX.URI.parse('xmpp:lisa@example.com?;item=1');
122    *   uri.queryParam('item');
123    *   // -> '1'
124    *
125    * @param {String} param The parameter who's value is looked up.
126    * @returns {String} The value of the parameter, or undefined if not found.
127    */
128   queryParam: function (param) {
129     if (!this.query) {
130       return undefined;
131     }
132 
133     var parts = this.query.split(';');
134     for (var i = 1, len = parts.length; i < len; i++) {
135       var kvp = parts[i].split('=');
136       if (kvp[0] === param) {
137         return kvp[1] || '';
138       }
139     }
140     return undefined;
141   },
142 
143   /**
144    * Convert URI object to string representation.
145    * @example
146    *   OX.URI.extend({ scheme: 'sip',
147    *                   path: 'lisa@example.com' }).toString();
148    *   //-> "sip:lisa@example.com"
149    * @returns {String} The URI converted to a string.
150    */
151   toString: function () {
152     var authority = this.authority ? '//' + this.authority + '/' : '',
153         query     = this.query     ? '?'  + this.query           : '',
154         fragment  = this.fragment  ? '#'  + this.fragment        : '';
155     return this.scheme + ':' + authority + this.path + query + fragment;
156   }
157 });
