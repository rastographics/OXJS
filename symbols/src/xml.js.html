  7 /**
  8  * @class
  9  * A simple XML element class.
 10  *
 11  * @example
 12  *   var newElement = OX.XML.Element.extend({name: 'foo'});
 13  *   newElement.attr('bar', 'bam');
 14  *   newElement.addChild(OX.XML.Element.extend({name: 'child'}));
 15  *   newElement.toString();
 16  *   // -> '<foo bar="bam"><child></child></foo>'
 17  *
 18  * @extends OX.Base
 19  */
 20 OX.XML.Element = OX.Base.extend(/** @lends OX.XML.Element# */{
 21   name: null,
 22   attributes: null,
 23   xmlns: null,
 24   children: null,
 25   text: null,
 26 
 27   /**
 28    * Get or set attributes on the receiver.
 29    *
 30    * @param {String} name The attributes name.
 31    * @param {String} [value] If value is supplied, the attribute will be set.
 32    * @returns {String} the value of the attribute.
 33    */
 34   attr: function (name, value) {
 35     this.attributes = this.attributes || {};
 36     if (value) {
 37       this.attributes[name] = value;
 38     }
 39     return this.attributes[name];
 40   },
 41 
 42   /**
 43    * Add a XML child element to the receiver.
 44    *
 45    * @param {OX.XML.Element} child The XML element to add as a child.
 46    * @returns {OX.XML.Element} The receiver.
 47    */
 48   addChild: function (child) {
 49     this.children = this.children || [];
 50     if (child) {
 51       this.children.push(child);
 52     }
 53     return this;
 54   },
 55 
 56   /**
 57    * @function
 58    * Escape XML characters to prevent parser errors.
 59    *
 60    * @param {String} string The string to escape.
 61    * @returns {String} The escaped string.
 62    */
 63   escapeXML: (function () {
 64     var character = {
 65       '"': '"',
 66       "'": ''',
 67       '<': '<',
 68       '>': '>',
 69       '&': '&'
 70     }, re = /[<>&"']/g;
 71     return function (str) {
 72       return str.replace(re, function (c) {
 73         return character[c];
 74       });
 75     };
 76   }()),
 77 
 78   /**
 79    * Return an XML string representation of this element.
 80    *
 81    * @returns {String} This XML element as XML text.
 82    */
 83   toString: function () {
 84     var ret = "";
 85     var attrs = [];
 86 
 87     if (this.xmlns) {
 88       this.attr('xmlns', this.xmlns);
 89     }
 90 
 91     if (this.attributes) {
 92       for (var name in this.attributes) {
 93         if (this.attributes.hasOwnProperty(name)) {
 94           var val = this.attributes[name];
 95           if (!val) {
 96             continue;
 97           }
 98 
 99           attrs.push(name + '="' + this.escapeXML(val.toString()) + '"');
100         }
101       }
102     }
103 
104     ret += "<" + this.name;
105     ret += (attrs.length > 0) ? ' ' + attrs.join(' ') : '';
106     ret += ">";
107 
108     var children = this.children || [];
109     for (var i = 0, len = children.length; i < len; i++) {
110       ret += this.children[i].toString();
111     }
112 
113     if (this.text) {
114       ret += this.escapeXML(this.text.toString());
115     }
116 
117     ret += "</" + this.name + ">";
118 
119     return ret;
120   }
121 }, /** @lends OX.XML.Element */ {
122 
123   /**
124    * Convenience function for creating a new OX.XML.Element and
125    * setting attrs and elements in a single function
126    *
127    * @param {Object} [attrs] A hash of attribute names to attribute values.
128    * @param {OX.XML.Element[]} [elements] An array of OX.XML.Element to assign as children.
129    * @returns {OX.XML.Element}
130    */
131   create: function (attrs, elements) {
132     var ret = this.extend();
133 
134     if (attrs) {
135       for (var k in attrs) {
136         if (attrs.hasOwnProperty(k)) {
137           var v = attrs[k];
138           if (!v) {
139             continue;
140           }
141           ret.attr(k, v);
142         }
143       }
144     }
145 
146     elements = (elements && elements.addChild) ? [elements] : elements;
147     if (elements && elements.length) {
148       for (var i = 0, len = elements.length; i < len; i++) {
149         ret.addChild(elements[i]);
150       }
151     }
152 
153     return ret;
154   }
155 });

278 /**
279  * @class
280  * XMPP XDataForm element.
281  *
282  * @extends OX.XML.Element
283  * @see <a href="http://xmpp.org/extensions/xep-0004.html">XEP-0004 Data Forms</a>
284  */
285 OX.XML.XMPP.XDataForm = OX.XML.Element.extend(/** @lends OX.XML.XMPP.XDataForm# */{
286   /** The Data Forms tag name */
287   name: 'x',
288   /** The Data Forms namespace */
289   xmlns: 'jabber:x:data',
290 
291   /**
292    * The 'type' attribute of the stanza.
293    *
294    * @param {String} [value] The value to set the 'type' attribute.
295    * @returns {String} The value of the 'type' attribute.
296    */
297   type: function (val) {
298     return this.attr('type', val);
299   },
300 
301   /**
302    * A convenience method for adding fields and values to the
303    * XDataForm. Calling this method will add an XDataField and value to
304    * this XDataForm.
305    *
306    * @param {String} name The name of the field, as identified in the 'var' attribute.
307    * @param {String} value The text to insert into the 'value' element.
308    * @param {String} type XDataField type see XEP: 0004.
309    * @returns {OX.XML.XMPP.XDataForm} The receiver.
310    */
311   addField: function (name, value, type) {
312     var f, v;
313     f = OX.XML.Element.extend({name: 'field'});
314     f.attr('var', name);
315 
316     if (value) {
317       v = OX.XML.Element.extend({name: 'value', text: value});
318       f.addChild(v);
319     }
320 
321     if (type) {
322       f.attr('type', type);
323     }
324 
325     return this.addChild(f);
326   }
