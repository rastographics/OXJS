1 /**
  2  * Utitity functions for OXJS.
  3  */
  4 OX.Base.mixin.call(Function.prototype, /** @scope Function.prototype */{
  5 
  6   /**
  7    * Marks the function as inferior.
  8    * If a key exists on the mixin, and the new mixin
  9    * is marked as inferior, it will ignore the new function,
 10    * relying on the old function for its implementation.
 11    * @example
 12    *   var foo = OX.Base.extend({
 13    *     bar: function () {
 14    *       return "superior";
 15    *     },
 16    *   }, {
 17    *     bar: function () {
 18    *       return "inferior";
 19    *     }.inferior()
 20    *   });
 21    *
 22    *   foo.bar();
 23    *   // -> "superior"
 24    * @returns {Function} The reciever.
 25    */
 26   inferior: function () {
 27     this._oxInferior = true;
 28     return this;
 29   },
 30 
 31   /**
 32    * <p>Provides access to the overridden base function.
 33    * If there is no base function to override, the first
 34    * argument to the function will be an empty function
 35    * that returns nothing.</p>
 36    *
 37    * <p>The first argument passed in is the base function,
 38    * bound to the correct scope (so you don't need to
 39    * call apply/call).</p>
 40    *
 41    * @example
 42    *   var foo = OX.Base.extend({
 43    *     bar: function (junk) {
 44    *       return 'bar' + junk;
 45    *     }
 46    *   });
 47    *   var fooBar = foo.extend({
 48    *     bar: function ($super, junk) {
 49    *       return 'foo' + $super(junk);
 50    *     }.around()
 51    *   });
 52    *
 53    *   foo.bar('bell');
 54    *   // -> 'barbell'
 55    *   fooBar.bar('n');
 56    *   // -> 'foobarn'
 57    * @returns {Function} The reciever.
 58    */
 59   around: function () {
 60     this._ox = this._ox || {};
 61 
 62     var empty = function () {},
 63     slice = Array.prototype.slice,
 64     /** @ignore */
 65     bind = function (lambda, that) {
 66       return function () {
 67         return lambda.apply(that, arguments);
 68       };
 69     };
 70 
 71     /** @ignore */
 72     this._ox.around = function (template, value, key) {
 73       var base = template[key] || empty;
 74       if (!(base instanceof Function)) {
 75         return value;
 76       }
 77 
 78       return function () {
 79         return value.apply(this, [bind(base, this)].concat(slice.apply(arguments)));
 80       };
 81     };
 82 
 83     return this;
 84   }
 85 });
