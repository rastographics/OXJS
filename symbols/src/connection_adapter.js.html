1 /**
  2  * @class
  3  * OX Connection Adapter abstract object.
  4  *
  5  * An instance of this object MUST be supplied to the OX.Connection
  6  * instance. This object is to be defined by consumers of the API as
  7  * an adapter to the XMPP connection library that is being used.
  8  * The example below shows a sample implementation for the
  9  * OX.ConnectionAdapter using the <a href="http://blog.jwchat.org/jsjac/">JSJaC</a> XMPP library.
 10  *
 11  * @example
 12  *   var conn = new JSJaCConnection({ httpbase: '/http-bind/' });
 13  *   var adapter = OX.ConnectionAdapter.extend({
 14  *     jid: function () { return conn.fulljid; },
 15  *
 16  *     registerHandler: function (event, handler) {
 17  *       return conn.registerHandler(event, handler);
 18  *     },
 19  *
 20  *     unregisterHandler: function (event, handler) {
 21  *       return conn.unregisterHandler(event, handler);
 22  *     },
 23  *
 24  *     send: function (xml, cb, args) {
 25  *       return conn._sendRaw(xml, cb, args);
 26  *     }
 27  *   });
 28  *
 29  *   var ox = OX.Connection.extend({ connectionAdapter: adapter });
 30  *
 31  * @extends OX.Base
 32  */
 33 OX.ConnectionAdapter = OX.Base.extend(/** @lends OX.ConnectionAdapter# */{
 34 
 35   /**
 36    * The JID of this connection.
 37    * @returns {String} The JID provided by your BOSH library.
 38    */
 39   jid: function () {
 40     throw new OX.Error("You MUST override OX.ConnectionAdapter#jid " +
 41                        "so it returns the JID of the BOSH connection.");
 42   },
 43 
 44   /**
 45    * Send an XML string to the underlying connection.
 46    *
 47    * @param {String} xml The XML String to send.
 48    * @param {Function} callback Called when a response to this packet is received with the first argument being the received packet.
 49    * @param {Array} args An array of arguments to be passed to callback after the packet.
 50    * @returns {void}
 51    *
 52    * @see OX.Connection#send
 53    */
 54   send: function (xml, callback, args) {
 55     throw new OX.Error("You MUST override OX.ConnectionAdapter#send " +
 56                        "so it will send XML over the BOSH connection.");
 57   },
 58 
 59   /**
 60    * Registers an event handler.
 61    *
 62    * @param {String} event The type of stanza for which to listen (i.e., `message', `iq', `presence.')
 63    * @param {Function} handler The stanza is passed to this function when it is received.
 64    * @returns {void}
 65    *
 66    * @see OX.ConnectionAdapter#unregisterHandler
 67    * @see OX.Connection#registerJIDHandler
 68    */
 69   registerHandler: function (event, handler) {
 70     throw new OX.Error("You MUST override OX.ConnectionAdapter#registerHandler " +
 71                        "so it will handle events the BOSH connection recieves.");
 72   },
 73 
 74   /**
 75    * Unregisters an event handler.
 76    *
 77    * @param {String} event The type of stanza we were listening to (i.e., `message', `iq', `presence.')
 78    * @returns {void}
 79    *
 80    * @see OX.ConnectionAdapter#registerHandler
 81    * @see OX.Connection#unregisterJIDHandler
 82    */
 83   unregisterHandler: function (event) {
 84     throw new OX.Error("You MUST override OX.ConnectionAdapter#unregisterHandler " +
 85                        "so it will remove event handlers on the BOSH connection.");
 86   }
 87 
 88 });
