1 /**
  2  * @namespace
  3  * <p>Namespace for user agent related services.</p>
  4  * <p>Active Calls pubsub allows for subscribing to evented
  5  * data regarding the current status of user agents in an OnSIP domain.</p>
  6  *
  7  * @extends OX.Base
  8  * @extends OX.Mixin.Subscribable
  9  * @requires A 'connectionAdapter' property inherited from an {@link OX.Connection}.
 10  * @see <a href="http://wiki.onsip.com/docs/User-Agent_Pubsub">UserAgent PubSub</a>
 11  */
 12 OX.Service.UserAgents = OX.Base.extend(OX.Mixin.Subscribable, /** @lends OX.Service.UserAgents */{
 13   /**
 14    * URI for this PubSub service.
 15    */
 16   pubSubURI: OX.Settings.URIs.pubSub.userAgents,
 17 
 18   /**
 19    * @namespace
 20    * User Agent Item.
 21    * @name OX.Service.UserAgents.Item
 22    * @extends OX.Item
 23    */
 24   Item: OX.Item.extend(/** @lends OX.Service.UserAgents.Item# */{
 25     /** The contact of this user agent. */
 26     contact:  null,
 27 
 28     /** The registration received IP & port. */
 29     received: null,
 30 
 31     /** The user agent identifier string. */
 32     device:   null,
 33 
 34     /** The time at which the user agent registration will expire. */
 35     expires:  null,
 36 
 37     /** The time at which a user-agent dialog event started */
 38     time: null
 39   }),
 40 
 41   /** @private
 42    * Returns an {@link OX.Service.UserAgents.Item} from an XML Document.
 43    * This method should be called once for each item to be constructed.
 44    * If a DOMElement contains more than one item node, only the first
 45    * item node will be returned as an {@link OX.Service.UserAgents.Item}
 46    *
 47    * @param {Element|Node} element The DOM Element or Node to parse into a {@link OX.Service.UserAgents.Item}
 48    * @returns {OX.Service.UserAgents.Item} The item created from the element passed in.
 49    */
 50   itemFromElement: function (element) {
 51     if (!element) {
 52       return undefined;
 53     }
 54 
 55     var userAgentNode = element.getElementsByTagName('user-agent'),
 56         attrs         = {connection: this.connection};
 57 
 58     if (!userAgentNode || !userAgentNode[0]) {
 59       return undefined;
 60     }
 61     var children = userAgentNode[0].childNodes;
 62 
 63     for (var i = 0, len = children.length; i < len; i++) {
 64       var node = children[i],
 65           value;
 66 
 67       if (!node.nodeName) {
 68         continue;
 69       }
 70 
 71       value = (node.firstChild && node.firstChild.nodeValue) || undefined;
 72       switch (node.nodeName.toLowerCase()) {
 73       case 'contact':
 74         attrs.contact = value;
 75         break;
 76       case 'received':
 77         attrs.received = value;
 78         break;
 79       case 'device':
 80         attrs.device = value;
 81         break;
 82       case 'expires':
 83         attrs.expires = value;
 84         break;
 85       case 'time':
 86         attrs.time = value;
 87         break;
 88       }
 89     }
 90 
 91     return this.Item.extend(attrs);
 92   }
 93 });
