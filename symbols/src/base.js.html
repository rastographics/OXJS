1 /**
  2  * @class
  3  * <p>Base object for OXJS. All other objects inherit from this one.
  4  * This provides object inheritance much like Douglas Crockford's
  5  * <a href="http://javascript.crockford.com/prototypal.html">Prototypal
  6  * Inheritance in JavaScript</a> with a few modifications of our own.</p>
  7  *
  8  * <p>This framework uses Object templates rather than classes to provide
  9  * inheritance.</p>
 10  */
 11 OX.Base = {
 12 
 13   /**
 14    * Creates a new object which extends the current object.  Any
 15    * arguments are mixed in to the new object as if {@link OX.Base.mixin}
 16    * was called on the new object with remaining args.
 17    *
 18    * @example
 19    *   var obj = OX.Base.extend({
 20    *     hello: "world"
 21    *   });
 22    *   obj.hello;
 23    *   // -> "world"
 24    * 
 25    *   OX.Base.hello;
 26    *   // -> undefined
 27    * @returns {OX.Base} the new object
 28    *
 29    * @see OX.Base.mixin
 30    */
 31   extend: function () {
 32     var F = function () {};
 33     F.prototype = this;
 34 
 35     var rc = new F();
 36     rc.mixin.apply(rc, arguments);
 37 
 38     if (rc.init && rc.init.constructor === Function) {
 39       rc.init.call(rc);
 40     }
 41     return rc;
 42   },
 43 
 44   /**
 45    * Iterates over all arguments, adding their own properties to the
 46    * receiver.
 47    *
 48    * @example
 49    *   obj.mixin({
 50    *     hello: "world"
 51    *   });
 52    *   obj.hello;
 53    *   // -> "world"
 54    *
 55    * @returns {OX.Base} the receiver
 56    *
 57    * @see OX.Base.extend
 58    */
 59   mixin: function () {
 60     var i, len, key, base, mixin, obj,
 61         /** @ignore */
 62         empty = function () {}, transform;
 63 
 64     for (i = 0, len = arguments.length; i < len; i++) {
 65       obj = arguments[i];
 66       for (key in obj) {
 67         if (obj.hasOwnProperty(key)) {
 68           mixin = obj[key];
 69 
 70           if (this[key] && mixin && mixin._oxInferior) {
 71             continue;
 72           }
 73 
 74           if (mixin && mixin instanceof Function && mixin._ox) {
 75             for (transform in mixin._ox) {
 76               if (mixin._ox.hasOwnProperty(transform)) {
 77                 mixin = mixin._ox[transform](this, mixin, key);
 78               }
 79             }
 80           }
 81 
 82           this[key] = mixin;
 83         }
 84       }
 85 
 86       // Prevents IE from clobbering toString
 87       if (obj && obj.toString !== Object.prototype.toString) {
 88         this.toString = obj.toString;
 89       }
 90     }
 91 
 92     return this;
 93   }
 94 };
