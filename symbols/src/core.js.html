1 /**
  2  * @namespace
  3  * <p>The core namespace for OX. OX is the official OnSIP XMPP-API JavaScript Library.</p>
  4  *
  5  * <p>For documentation concerning the OnSIP XMPP-API itself,
  6  * visit the <a href="http://wiki.onsip.com/docs/XMPP_API">wiki</a>.</p>
  7  * @version 2.0.1
  8  */
  9 var OX = {
 10 
 11   /**
 12    * The version number for OX using <a href="http://semver.org">Semantic Versioning</a>.
 13    * @type String
 14    */
 15   VERSION: '2.1.1',
 16 
 17   /**
 18    * Prints a debug message to the console, if window.console exists.
 19    * @returns {void}
 20    */
 21   debug: function () {
 22     return window.console && window.console.debug && window.console.debug.apply && window.console.debug.apply(window.console, arguments);
 23   },
 24 
 25   /**
 26    * Prints a log to the console, if window.console exists.
 27    * @returns {void}
 28    */
 29   log: function () {
 30     return window.console && window.console.log && window.console.log.apply && window.console.log.apply(window.console, arguments);
 31   },
 32 
 33   /**
 34    * Prints a warning to the console, if window.console exists.
 35    * @returns {void}
 36    */
 37   warn: function () {
 38     return window.console && window.console.warn && window.console.warn.apply && window.console.warn.apply(window.console, arguments);
 39   },
 40 
 41   /**
 42    * Prints an error to the console, if window.console exists.
 43    * @returns {void}
 44    */
 45   error: function () {
 46     return window.console && window.console.error && window.console.error.apply && window.console.error.apply(window.console, arguments);
 47   },
 48 
 49   /**
 50    * Begins a console group, if it's able to; otherwise it tries to print a log.
 51    * @returns {void}
 52    */
 53   group: function () {
 54     if (window.console && window.console.group) {
 55       window.console.group.apply(window.console, arguments);
 56     } else {
 57       OX.log.apply(OX, arguments);
 58     }
 59   },
 60 
 61   /**
 62    * End a console group.
 63    * @returns {void}
 64    */
 65   groupEnd: function () {
 66     if (window.console && window.console.groupEnd) {
 67       window.console.groupEnd();
 68     }
 69   }
 70 
 71 };
