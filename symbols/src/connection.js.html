1 /**
  2  * @class
  3  * <p>Connection object to use for all OXJS connections.
  4  * The connection MUST have a valid connectionAdapter to function.
  5  * The Connection object is the core API for OXJS.</p>
  6  *
  7  * <p>The Connection assumes that the consumer understands when
  8  * to make calls (when the connection is established). This allows
  9  * you to configure OX at read-time, rather than at run-time.</p>
 10  *
 11  * @example
 12  *   var bosh = new Strophe.Connection('/http-bind/');
 13  *   var ox = OX.Connection.extend({
 14  *     connectionAdapter: OX.StropheAdapter.extend({
 15  *       connection: bosh
 16  *     })
 17  *   });
 18  *
 19  *   // A naive approach...
 20  *   bosh.connect("juliet@example.com", "romeo", function (status) {
 21  *     if (status === Strophe.Status.CONNECTED) {
 22  *       // OK- OX is ready to go! Let's authenticate...
 23  *       ox.Auth.authorizePlain("sip@example.com", "password", "jid@example.com", {
 24  *         onSuccess: function () {
 25  *           alert("You got the OK to go on ahead!");
 26  *         }
 27  *       });
 28  *     }
 29  *   });
 30  *
 31  * @extends OX.Base
 32  * @property {OX.Service.Auth} Auth A usable instance of the Auth service.
 33  * @property {OX.Service.ActiveCalls} ActiveCalls A useable instance of the ActiveCalls service.
 34  * @property {OX.Service.UserAgents} UserAgents A usable instance of the UserAgents service.
 35  * @property {OX.Service.Voicemail} Voicemail A usable instance of the Voicemail service.
 36  * @property {OX.Service.Directories} Directories A usable instance of the Directories service.
 37  * @property {OX.Service.Preferences} Preferences A usable instance of the Preferences service (not implemented).
 38  * @property {OX.Service.RecentCalls} RecentCalls A usable instance of the RecentCalls service (not implemented).
 39  * @requires A 'connectionAdapter' property that extends {@link OX.ConnectionAdapter}.
 40  */
 41 OX.Connection = OX.Base.extend(/** @lends OX.Connection# */{
 42 
 43   /**
 44    * @private
 45    * Map of instance names to instance objects.
 46    * They are instantiated on extend time when a
 47    * connectionAdapter is provided.
 48    */
 49   services: {
 50     Auth:        OX.Service.Auth,
 51     ActiveCalls: OX.Service.ActiveCalls,
 52     Directories: OX.Service.Directories,
 53     Preferences: OX.Service.Preferences,
 54     RecentCalls: OX.Service.RecentCalls,
 55     UserAgents:  OX.Service.UserAgents,
 56     Voicemail:   OX.Service.Voicemail,
 57     Rosters:     OX.Service.Rosters
 58   },
 59 
 60   /**
 61    * @private
 62    * Map of jids to event handler functions. Used when message events
 63    * are received from the connection.
 64    *
 65    * @see OX.Connection#registerJIDHandler
 66    * @see OX.Connection#unregisterJIDHandler
 67    */
 68   jidHandlers: {},
 69 
 70   /**
 71    * @private
 72    * Initialize the Connection with Services hooked up
 73    * with a connection and then stuck on the top level namespace.
 74    * @param {Function} $super The base init that was implemented under {@link OX.Connection#init}
 75    * @returns {void}
 76    */
 77   init: function ($super) {
 78     if (this.connectionAdapter) {
 79       var serviceMap = {};
 80 
 81       for (var s in this.services) {
 82         if (this.services.hasOwnProperty(s)) {
 83           var service = this.services[s];
 84 
 85           this[s] = service.extend({connection: this});
 86           if (service.pubSubURI) {
 87             serviceMap[service.pubSubURI] = service;
 88           }
 89         }
 90       }
 91 
 92       // Register for incoming messages.
 93       var that = this;
 94       this.connectionAdapter.registerHandler('message', function (msg) {
 95         var from = msg.getFrom(),
 96             fn = that.jidHandlers[from];
 97         if (fn) {
 98           fn(msg);
 99         }
100       });
101     }
102 
103     $super();
104   }.around(),
105 
106   /**
107    * Sends an XML string to the connection adapter.
108    *
109    * @param {String} xml The XML String to send.
110    * @param {Function} callback Called when a response to this packet is received with the first argument being the received packet.
111    * @param {Array} [args] An array of arguments to be passed to callback after the packet.
112    * @returns {void}
113    *
114    * @see OX.ConnectionAdapter#send
115    */
116   send: function (xml, callback, args) {
117     this.connectionAdapter.send(xml, callback, args || []);
118   },
119 
120   /**
121    * Returns the JID of this connection.
122    *
123    * @example
124    *   ox.getJID();
125    *   // -> "mock@example.com/some-resource-identifier"
126    *
127    * @returns {String} This connection's JID.
128    *
129    * @see OX.ConnectionAdapter#jid
130    */
131   getJID: function () {
132     return this.connectionAdapter.jid();
133   },
134 
135   /**
136    * Registers a message event handler for a JID. Only one
137    * handler is active at a time per JID.
138    *
139    * @param {String} jid The jid who's events we listen to.
140    * @param {Function} handler Function of one argument: the message packet received.
141    * @param {Object} [target] The optional target to associate 'this' with.
142    * @returns {OX.Connection} The receiver.
143    * @example
144    *   var ox = OX.Connection.extend({ connectionAdapter: bosh });
145    *   ox.registerJIDHandler('pubsub.active-calls.xmpp.onsip.com', function (packet) {
146    *     ...
147    *   });
148    *
149    * @see OX.Connection#registerJIDHandler
150    * @see OX.ConnectionAdapter#registerHandler
151    */
152   registerJIDHandler: function (jid, handler, target) {
153     this.jidHandlers[jid] = function () {
154       target = target || this;
155       return handler.apply(target, arguments);
156     };
157     return this;
158   },
159 
160   /**
161    * Unregister the handler, if any, for a JID.
162    *
163    * @example
164    *   var ox = OX.Connection.extend({ connectionAdapter: bosh });
165    *   ox.unregisterJIDHandler('pubsub.active-calls.xmpp.onsip.com');
166    *
167    * @param {String} jid The jid who's events we listen to.
168    * @returns {OX.Connection} The receiver.
169    *
170    * @see OX.Connection#unregisterJIDHandler
171    * @see OX.ConnectionAdapter#unregisterHandler
172    */
173   unregisterJIDHandler: function (jid) {
174     delete this.jidHandlers[jid];
175     return this;
176   }
177 });
