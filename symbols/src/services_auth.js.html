1 /**
  2  * @namespace
  3  * <p>Namespace for auth related services.</p>
  4  * <p>In order to allow any JID to perform a command for a SIP address
  5  * or subscribe to PubSub data regarding a SIP address or SIP domain the
  6  * JID must authorize itself (or another JID) by providing a SIP user
  7  * address and a password to the Auth component.</p>
  8  *
  9  * @extends OX.Base
 10  * @requires A 'connectionAdapter' property inherited from an {@link OX.Connection}.
 11  * @see <a href="http://wiki.onsip.com/docs/Auth_Component">Auth Component</a>
 12  */
 13 OX.Service.Auth = OX.Base.extend(OX.Mixin.EntityTime, /** @lends OX.Service.Auth */{
 14 
 15   /**
 16    * Requests the time from the authentication service.
 17    * Used to get an authoritative time source so it's known
 18    * when you have to re-authenticate with the XMPP API.
 19    *
 20    * @param {Object} [callbacks] Callbacks with 'onSuccess' and 'onError'
 21    *   @param {Function} [callbacks.onSuccess] The success callback
 22    *     @param {OX.PacketAdapter} [callbacks.onSuccess.packet] The packet recieved.
 23    *     @param {Object} [callbacks.onSuccess.time] The parsed time with slots 'tzo' and 'utc'.
 24    *   @param {Function} [callbacks.onError] The error callback
 25    *     @param {OX.PacketAdapter} [callbacks.onError.packet] The packet recieved.
 26    * @returns {void}
 27    */
 28   entityTime: function (cb) {
 29     return this.getTime(OX.Settings.URIs.entity.auth, cb);
 30   },
 31 
 32   /**
 33    * Authorize a JID for a SIP address, authorized via a password. This
 34    * password is sent in clear text to the XMPP API, so your connection
 35    * should be encrypted for your own safety.
 36    *
 37    * @param {String} address The SIP address for authentication.
 38    * @param {String} password The web password for the SIP address.
 39    * @param {Object} [callbacks] An object supplying functions for 'onSuccess', and 'onError'.
 40    * @param {Object} [options]
 41    *   @param {String} options.jid The JID to authorize for the SIP address. If unspecified, use the current JID from the underlying connection.
 42    *   @param {String} options.authForAll Authorize this JID for all SIP addresses associated with it.
 43 
 44    * @returns {void}
 45    *
 46    * @example
 47    *   ox.Auth.authorizePlain('sip@example.com', 'password', {
 48    *     onSuccess: function () {},
 49    *     onError:   function (error) {}
 50    *   }, { jid: 'jid@example.com', authForAll: true });
 51    */
 52   authorizePlain: function (address, password, callbacks, options) {
 53     var iq    = OX.XML.XMPP.IQ.extend(),
 54         cmd   = OX.XML.XMPP.Command.extend(),
 55         xData = OX.XML.XMPP.XDataForm.extend(),
 56         uri   = OX.Settings.URIs.command.authorizePlain,
 57         authForAll, jid;
 58 
 59     options = options || {};
 60     callbacks = callbacks || {};
 61     authForAll = !!options.authForAll;
 62     jid = options.jid;
 63 
 64     iq.to(uri.path);
 65     iq.type('set');
 66     cmd.node(uri.queryParam('node'));
 67     xData.type('submit');
 68     xData.addField('sip-address', address);
 69     xData.addField('password', password);
 70     xData.addField('auth-for-all', authForAll ? 'true' : 'false');
 71     if (jid) {
 72       xData.addField('jid', jid);
 73     }
 74     iq.addChild(cmd.addChild(xData));
 75 
 76     this.connection.send(iq.toString(), function (packet) {
 77       if (!packet) {
 78         return;
 79       }
 80 
 81       if (packet.getType() === 'error' && callbacks.onError) {
 82         callbacks.onError(packet);
 83       } else if (callbacks.onSuccess) {
 84         callbacks.onSuccess(packet);
 85       }
 86     }, []);
 87   }
 88 });
