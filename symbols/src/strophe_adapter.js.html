1 /**
  2  * @class
  3  * Provides a Connection Adapter for the BOSH library
  4  * <a href="http://code.stanziq.com/strophe">Strophe</a>.
  5  * The adapter tries to deal with possible memory leaks due to unanswered
  6  * IQs, having a max queue size of handlers. You should be able to plug-and-play
  7  * with this adapter and Strophe like the following example:
  8  *
  9  * @example
 10  *   var bosh = new Strophe.Connection('/http-bind/');
 11  *   var ox = OX.Connection.extend({
 12  *     connectionAdapter: OX.StropheAdapter.extend({
 13  *       connection: bosh
 14  *     })
 15  *   });
 16  * @extends OX.ConnectionAdapter
 17  * @requires A slot named "connection" with an instance of Strophe.Connection provided.
 18  */
 19 OX.StropheAdapter = OX.ConnectionAdapter.extend(
 20   /** @lends OX.StropheAdapter# */{
 21 
 22   /** @private */
 23   _callbacks: {},
 24   /** @private */
 25   _handlers: {},
 26   /** @private */
 27   _callbackQueue: [],
 28 
 29   /**
 30    * <p>The maximum allowable size for the callback queue.
 31    * When it reaches the maximum size, it will warn you about it,
 32    * and begin removing stale handlers, assuming that they will never be called.
 33    * This exists as a catch for memory leaks. Change this value to meet your needs.</p>
 34    *
 35    * <p>You <i>will</i> be warned when this quota is reached.
 36    * Make sure that you aren't throwing away any live messages
 37    * if you want to keep the MAX_QUEUE_SIZE where it is.</p>
 38    */
 39   MAX_QUEUE_SIZE: 100,
 40 
 41   /** @private */
 42   init: function ($super) {
 43     this._callbacks = {};
 44     this._handlers = {};
 45     this._callbackQueue = [];
 46     $super();
 47   }.around(),
 48 
 49   /**
 50    * The connection's JID.
 51    * @returns {String} The JID associated with the connection.
 52    */
 53   jid: function () {
 54     return this.connection.jid;
 55   },
 56 
 57   /**
 58    * Subscribe to stanza via top-level XMPP tag name.
 59    *
 60    * @param {String} event The top level XMPP tag name to register for.
 61    * @param {Function} handler The function handler for the event.
 62    * @returns {void}
 63    */
 64   registerHandler: function (event, handler) {
 65     var that = this;
 66 
 67     var wrapper = function (stanza) {
 68       var packetAdapter = that.toPacket(stanza),
 69           newArgs = [packetAdapter];
 70 
 71       for (var i = 1, len = arguments.length; i < len; i++) {
 72         newArgs.push(arguments[i]);
 73       }
 74 
 75       try {
 76         handler.apply(this, newArgs);
 77       } catch (e) {
 78         OX.error('Error in OX handler: ' + handler +
 79                  '; Error: ' + e + '; response stanza: ' + stanza);
 80         throw e;
 81       }
 82       return true;
 83     };
 84 
 85     this.unregisterHandler(event);
 86     this._handlers[event] = this.connection.addHandler(wrapper, null, event,
 87                                                        null, null, null);
 88   },
 89 
 90   /**
 91    * Unsubscribe from corresponding event.
 92    *
 93    * @param {String} event The event to unsubscribe from.
 94    * @returns {void}
 95    */
 96   unregisterHandler: function (event) {
 97     var queue = this._callbackQueue, i, len = queue.length, rest;
 98 
 99     if (this._handlers[event]) {
100       this.connection.deleteHandler(this._handlers[event]);
101       delete this._handlers[event];
102 
103       // Remove from the callback queue
104       for (i = 0; i < len; i++) {
105         if (queue[i] === event) {
106           rest = queue.slice(i + 1, queue.length);
107           queue.length = i;
108           queue.push(rest);
109           break;
110         }
111       }
112     }
113   },
114 
115   /**
116    * Create a DOM node via XML.
117    *
118    * @private
119    * @param {String} xml The xml string to convert into an object.
120    * @returns {Element} The document fragment that represents the xml string.
121    */
122   createNode: function (xml) {
123     var node = null, parser = null;
124     if (window.ActiveXObject) {
125       parser = new ActiveXObject("Microsoft.XMLDOM");
126       parser.async = "false";
127       parser.setProperty("SelectionLanguage", "XPath");
128       parser.loadXML(xml);
129       node = parser.firstChild;
130     } else {
131       parser = new DOMParser();
132       node = parser.parseFromString(xml, 'text/xml');
133       node = node.firstChild;
134       document.adoptNode(node);
135     }
136     return node;
137   },
138 
139   /**
140    * Send the xml fragment over the connection.
141    *
142    * @param {String} xml The xml to send.
143    * @param {Function} callback The function to call when done.
144    * @param {Array} args A list of arguments to provide to the callback.
145    * @returns {void}
146    */
147   send: function (xml, callback, args) {
148     var node = this.createNode(xml),
149         that = this;
150 
151     if (!this.connection.connected || this.connection.disconnecting) {
152       OX.log('Prevented "' + xml + '" from being sent because ' +
153              'the BOSH connection is being disposed / is disposed.');
154       return false;
155     }
156 
157     if (callback) {
158       var wrapper = function (stanza) {
159         var packetAdapter = that.toPacket(stanza),
160             newArgs = [packetAdapter],
161             queue = that._callbackQueue, rest,
162             event = node.getAttribute('id'), i, len;
163         args = args || [];
164         for (i = 0, len = args.length; i < len; i++) {
165           newArgs.push(args[i]);
166         }
167 
168         // Remove from the callback queue
169         for (i = 0, len = queue.length; i < len; i++) {
170           if (queue[i].toString() === event) {
171             rest = queue.slice(i + 1, queue.length);
172             queue.length = i;
173             queue.push.apply(queue, rest);
174             break;
175           }
176         }
177 
178         delete that._callbacks[event];
179 
180         try {
181           callback.apply(this, newArgs);
182         } catch (e) {
183           OX.error('Error in OX handler: ' + callback +
184                    '; Error: ' + e + '; response stanza: ' + stanza);
185           throw e;
186         }
187 
188         return false;
189       };
190 
191       var id = node.getAttribute('id');
192       if (!id) {
193         id = this.connection.getUniqueId();
194         node.setAttribute('id', id);
195       }
196 
197       this._callbacks[id] = this.connection.addHandler(wrapper, null, null,
198                                                        null, id, null);
199       this._callbackQueue.unshift(id);
200       if (this._callbackQueue.length > this.MAX_QUEUE_SIZE) {
201         OX.warn("You have too many callbacks waiting for a response, so I'm getting rid of the oldest one.\n" +
202                 "If this isn't desired, override the MAX_QUEUE_SIZE of OX.StropheAdapter.");
203         delete this._callbacks[this._callbackQueue.pop()];
204       }
205     }
206     return this.connection.send(node);
207   },
208 
209   /**
210    * @private
211    * Convert a stanza into an object that implements {@link OX.PacketAdapter}.
212    *
213    * @param {Element} stanza The XMPP stanza to pack.
214    *
215    * @returns {OX.PacketAdapter} The stanza wrapped as a packet.
216    */
217   toPacket: function (stanza) {
218     var to = stanza.getAttribute('to'),
219         from = stanza.getAttribute('from'),
220         type = stanza.getAttribute('type');
221     return {
222       getFrom: function () {
223         return from;
224       },
225       getType: function () {
226         return type;
227       },
228       getTo: function () {
229         return to;
230       },
231       getNode: function () {
232         return stanza;
233       }
234     };
235   }
236 });
