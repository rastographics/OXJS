1 /**
  2  * @namespace
  3  * Entity Time Mixin.
  4  *
  5  * @see <a href="http://xmpp.org/extensions/xep-0202.html">XEP 0202: Entity Time</a>
  6  * @requires A 'connectionAdapter' property which is an {@link OX.ConnectionAdapter} object on receiving object.
  7  */
  8 OX.Mixin.EntityTime = /** @lends OX.Mixin.EntityTime# */{
  9 
 10   /**
 11    * Request the time from the XMPP server you're connected to.
 12    * @param {OX.URI} entityURI The URI to query for the time.
 13    * @param {Object} [callbacks] Callbacks with 'onSuccess' and 'onError'
 14    *   @param {Function} [callbacks.onSuccess] The success callback
 15    *     @param {OX.PacketAdapter} [callbacks.onSuccess.packet] The packet recieved.
 16    *     @param {Object} [callbacks.onSuccess.time] The parsed time with slots 'tzo' and 'utc'.
 17    *       @param {String} [callbacks.onSuccess.time.utc] The UTC time according to the responding entity.
 18    *       @param {String} [callbacks.onSuccess.time.tzo] The entity's numeric time zone offset from UTC.
 19    *   @param {Function} [callbacks.onError] The error callback
 20    *     @param {OX.PacketAdapter} [callbacks.onError.packet] The packet recieved.
 21    * @returns {void}
 22    */
 23   getTime: function (entityURI, callbacks) {
 24     var iq = OX.XML.XMPP.IQ.extend(),
 25         time = OX.XML.Element.extend({name: 'time',
 26                                       xmlns: 'urn:xmpp:time'});
 27 
 28     iq.to(entityURI.path);
 29     iq.type('get');
 30     iq.addChild(time);
 31 
 32     this.connection.send(iq.toString(), function (packet) {
 33       if (packet.getType() === 'error' && callbacks.onError) {
 34         callbacks.onError(packet);
 35       } else if (callbacks.onSuccess) {
 36         var node = packet.getNode(),
 37             elTZO = node.getElementsByTagName('tzo')[0],
 38             elUTC = node.getElementsByTagName('utc')[0];
 39 
 40         callbacks.onSuccess(packet, {
 41           tzo: elTZO && (elTZO.textContent || elTZO.text),
 42           utc: elUTC && (elUTC.textContent || elUTC.text)
 43         });
 44       }
 45     });
 46   }
 47 };
