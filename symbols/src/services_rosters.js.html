1 /**
  2  * @namespace
  3  * <p>Namespace for roster related services.</p>
  4  * <p>This service provides a way to request a Roster Item Exchange
  5  * that advises an XMPP Client what to do with the user's Roster
  6  * (adding, modifying, and deleting entities).</p>
  7  * @extends OX.Base
  8  * @requires A 'connectionAdapter' property inherited from an {@link OX.Connection}.
  9  * @see <a href="http://wiki.onsip.com/docs/Rosters_Component">Rosters Component</a>
 10  * @see <a href="http://xmpp.org/extensions/xep-0144.html">XEP-0144: Roster Item Exchange</a>
 11  */
 12 OX.Service.Rosters = OX.Base.extend(OX.Mixin.Subscribable, /** @lends OX.Service.Rosters */{
 13   /**
 14    * Push a roster group from the Junction Networks XMPP API Rosters Component.
 15    * The first time this is called, a user will receive a series of roster add requests for
 16    * every user in his organization. The next time he requests roster information he will only
 17    * receive deltas; that is, add requests of any new users since his last request,
 18    * modify requests for any user's who have changed contact information, and
 19    * delete requests for any users who may have been deleted.
 20    *
 21    * @param {Object} [callbacks] Callbacks with 'onSuccess' and 'onError'
 22    *   @param {Function} [callbacks.onSuccess] The success callback
 23    *     @param {OX.PacketAdapter} [callbacks.onSuccess.packet] The packet recieved.
 24    *   @param {Function} [callbacks.onError] The error callback
 25    *     @param {OX.PacketAdapter} [callbacks.onError.packet] The packet recieved.
 26    * @param {Object} [options]
 27    *   @param {String} options.jid The full JID to push roster groups to; if not provided, the JID in the IQ 'from' attribute will be assumed.
 28    * @returns {void}
 29    *
 30    * @example
 31    * ox.Rosters.pushRosterGroups({
 32    *   onSuccess: function () {},
 33    *   onError:   function (error) {}
 34    * }, { jid: 'jid@example.com' });
 35    */
 36   pushRosterGroups: function (callbacks, options) {
 37     var iq    = OX.XML.XMPP.IQ.extend(),
 38         cmd   = OX.XML.XMPP.Command.extend(),
 39         xData = OX.XML.XMPP.XDataForm.extend(),
 40         uri   = OX.Settings.URIs.command.pushRosterGroups,
 41         jid;
 42 
 43     callbacks = callbacks || {};
 44     jid = options && options.jid;
 45 
 46     iq.to(uri.path);
 47     iq.type('set');
 48     cmd.node(uri.queryParam('node'));
 49     xData.type('submit');
 50     if (jid) {
 51       xData.addField('jid', jid);
 52     }
 53 
 54     iq.addChild(cmd.addChild(xData));
 55 
 56     this.connection.send(iq.toString(), function (packet) {
 57       if (!packet) {
 58         return;
 59       }
 60 
 61       if (packet.getType() === 'error' && callbacks.onError) {
 62         callbacks.onError(packet);
 63       } else if (callbacks.onSuccess) {
 64         callbacks.onSuccess(packet);
 65       }
 66     }, []);
 67   }
 68 });
