  2  * @namespace
  3  * Subscribable mixin.
  4  * Something's subscribable if it communicates with a JID that supports
  5  * <a href="http://xmpp.org/extensions/xep-0060.html">XEP-0060: Publish-Subscribe</a>.
  6  *
  7  * @requires A 'connectionAdapter' property which is an {@link OX.ConnectionAdapter} object on receiving object.
  8  * @requires A 'pubSub' property that's the URI of the PubSub service.
  9  * @requires An implementation of 'itemFromElement' which takes an Node or Element and returns an item.

 20   function getFirstElementChild(el) {
 21     var nodeType = ELEMENT_NODE;
 22     for (var i = 0, l = el.childNodes.length; i < l; i++) {
 23       if (el.childNodes[i].nodeType === nodeType) {
 24         return el.childNodes[i];
 25       }
 26     }
 27     return null;
 28   }

 30   function packetType(element) {
 31     if (!element) {
 32       return null;
 33     }
 34 
 35     switch (element.tagName) {
 36     case 'subscription':
 37       return element.getAttribute('subscription');
 38     case 'items':
 39       if (getFirstElementChild(element).tagName === 'retract') {
 40         return 'retract';
 41       }
 42       return 'publish';
 43     default:
 44       return null;
 45     }
 46   }

 48   function convertItems(document) {
 49     function itemURI(itemID, node) {
 50       var from  = document.getAttribute('from'),
 51           items = getFirstElementChild(getFirstElementChild(document));
 52 
 53       return OX.URI.fromObject({path: from,
 54                                 query: ';node=' + node + ';item=' + itemID});
 55     }
 56     function getPublishTime(element) {
 57       var firstChild  = getFirstElementChild(element),
 58           publishTime = firstChild && firstChild.getAttribute('publish-time');
 59 
 60       return publishTime;
 61     }
 62 
 63     /*
 64      * TODO: Without XPath we're taking some schema risks
 65      * here. Really we only want `/iq/pubsub/items/item'
 66      * nodes. Since we can't do that easily, just grab any `items'
 67      * elements and pass any immediate descendants named `item' to
 68      * itemFromElement.
 69      */
 70     var rc    = [],
 71         items = document.getElementsByTagName('items') || [];
 72 
 73     // Grab the first `items' node found.
 74     for (var i = 0, len = items.length; i < len; i++) {
 75       if (items[i] && items[i].childNodes) {
 76         var children = items[i].childNodes,
 77             node     = items[i].getAttribute('node') || '/',
 78             item;
 79 
 80         for (var ii = 0, ilen = children.length; ii < ilen; ii++) {
 81           if (children[ii].tagName && children[ii].tagName === 'item') {
 82             item = this.itemFromElement(children[ii]);
 83 
 84             item.publishTime = getPublishTime(children[ii]);
 85             item.uri = itemURI(children[ii].getAttribute('id'),
 86                                node);
 87             rc.push(item);
 88           }
 89         }
 90       }
 91     }
 92 
 93     return rc;
 94   }

 96   function redirectURIFromPacket(packet) {
 97     var e = packet && packet.getNode().getElementsByTagName('error')[0],
 98         fChild = e && getFirstElementChild(e),
 99         tName = fChild && fChild.tagName,
100         uri = fChild && fChild.firstChild && fChild.firstChild.nodeValue;
101 
102     return packet && packet.getType() === 'error' &&
103            (tName === 'redirect' || tName === 'gone') &&
104            uri && OX.URI.parse(uri) || null;
105   }

107   function pubSubPacketHandler(actions, packet, node, options, that) {
108     if (!packet) {
109       return;
110     }
111     that = that || this;
112     var optDefaults = {
113       redirectCount: 0,
114       origURI: node ? that.pubSubURI.extend({query: ';node=' + node}): that.pubSubURI // this will only be set the first time
115     }, redirectURI = redirectURIFromPacket(packet);
116 
117     options = OX.Base.mixin.call(optDefaults, options, {
118       finalURI: node ? that.pubSubURI.extend({query: ';node=' + node}): that.pubSubURI // this will be set every time
119     });
120 
121     if (packet.getType() === 'result') {
122       actions.result.call(that, packet, node, options);
123     } else if (options.redirectCount < MAX_REDIRECTS &&
124                redirectURI && redirectURI.path &&
125                redirectURI.queryParam('node')) {
126 
127       options.redirectCount += 1;
128       actions.redirect.call(that, packet, redirectURI.queryParam('node'), options);
129     } else {
130       actions.error.call(that, packet, node, options);
131     }
132   }

134   function fireEvent(type, packet) {
135     function subscriptionURI() {
136       var elt    = packet.getNode(),
137           from   = elt.getAttribute('from'),
138           sub    = getFirstElementChild(getFirstElementChild(elt)),
139           node   = sub.getAttribute('node') || '/';
140 
141       return OX.URI.fromObject({path:   from, query: ';node=' + node});
142     }
143 
144     function retractURI() {
145       var elt    = packet.getNode(),
146           from   = elt.getAttribute('from'),
147           items  = elt.getElementsByTagName('items')[0],
148           node   = items.getAttribute('node') || '/',
149           itemID = getFirstElementChild(items).getAttribute('id');
150 
151       return OX.URI.fromObject({path:  from,
152                                 query: ';node=' + node + ';item=' + itemID});
153     }
154 
155     switch (type) {
156     case 'subscribed':
157       if (this._subscriptionHandlers.onSubscribed) {
158         var subscribedURI = subscriptionURI();
159         this._subscriptionHandlers.onSubscribed(subscribedURI, packet);
160       }
161       break;
162     case 'pending':
163       if (this._subscriptionHandlers.onPending) {
164         var pendingURI = subscriptionURI();
165         this._subscriptionHandlers.onPending(pendingURI, packet);
166       }
167       break;
168     case 'none':
169       if (this._subscriptionHandlers.onUnsubscribed) {
170         var unsubscribedURI = subscriptionURI();
171         this._subscriptionHandlers.onUnsubscribed(unsubscribedURI, packet);
172       }
173       break;
174     case 'publish':
175       if (this._subscriptionHandlers.onPublish) {
176         var items = convertItems.call(this, packet.getNode());
177         for (var i = 0, len = items.length; i < len; i++) {
178           this._subscriptionHandlers.onPublish(items[i], packet);
179         }
180       }
181       break;
182     case 'retract':
183       if (this._subscriptionHandlers.onRetract) {
184         this._subscriptionHandlers.onRetract(retractURI(), packet);
185       }
186       break;
187     }
188   }

190   function jidHandler(packet) {
191     var event = packet.getNode().getElementsByTagName('event')[0];
192     if (!event) {
193       return;
194     }
195 
196     fireEvent.call(this, packetType(getFirstElementChild(event)), packet);
197   }

199   function getSubscriptionsHandler(packet, node, callbacks, options) {
200     callbacks = OX.Base.mixin.call(callbacks, cbDefault);
201 
202     pubSubPacketHandler({
203       result: function (packet, node, options) {
204         var subscriptions = [],
205             subElements = packet.getNode().getElementsByTagName('subscription');
206         for (var i = 0; i < subElements.length; i++) {
207           if (options.strict && this.connection.getJID() !== subElements[i].getAttribute('jid')) {
208             continue;
209           }
210 
211           subscriptions.push({
212             node: subElements[i].getAttribute('node'),
213             jid: subElements[i].getAttribute('jid'),
214             subscription: subElements[i].getAttribute('subscription'),
215             subid: subElements[i].getAttribute('subid')
216           });
217         }
218         callbacks.onSuccess(options.origURI, options.finalURI, subscriptions, packet);
219       },
220       redirect: function (packet, node, options) {
221         doGetSubscriptions.call(this, node, callbacks, options);
222       },
223       error: function (packet, node, options) {
224         callbacks.onError(options.origURI, options.finalURI, packet);
225       }
226     }, packet, node, options, this);
227 
228   }

247   function subscriptionHandler(packet, node, subOptions, callbacks, options) {
248     callbacks = OX.Base.mixin.call(callbacks, cbDefault);
249 
250     pubSubPacketHandler({
251       result: function (packet, node, options) {
252         callbacks.onSuccess(options.origURI, options.finalURI, packet);
253         var pubSub = packet.getNode().getElementsByTagName('pubsub')[0] || {},
254             subscription = getFirstElementChild(pubSub);
255 
256         fireEvent.call(this, packetType(subscription), packet);
257       },
258       redirect: function (packet, node, options) {
259         doSubscribe.call(this, node, subOptions, callbacks, options);
260       },
261       error: function (packet, node, options) {
262         callbacks.onError(options.origURI, options.finalURI, packet);
263       }
264     }, packet, node, options, this);
265   }

267   function unsubscriptionHandler(packet, node, callbacks) {
268     var uri = this.pubSubURI.extend({query: ';node=' + node});
269     callbacks = callbacks || {};
270 
271     if (!packet) {
272       return;
273     }
274 
275     if (packet.getType() === 'error') {
276       if (callbacks.onError) {
277         callbacks.onError(uri, packet.getNode());
278       }
279     } else {
280       if (callbacks.onSuccess) {
281         callbacks.onSuccess(uri, packet.getNode());
282       }
283     }
284   }

286   function getItemsHandler(packet, callbacks) {
287     callbacks = callbacks || {};
288 
289     if (!packet) {
290       return;
291     }
292 
293     if (packet.getType() === 'error') {
294       if (callbacks.onError) {
295         callbacks.onError(packet);
296       }
297     } else {
298       if (callbacks.onSuccess) {
299         callbacks.onSuccess(convertItems.call(this, packet.getNode()));
300       }
301     }
302   }

304   function zeroPad(spaces, value) {
305     var rc = (value || '').toString();
306     for (var i = spaces - rc.length; i > 0; i--) {
307       rc = '0' + rc;
308     }
309     return rc;
310   }

312   var optionTransforms = {
313     expire: function (direction, value) {
314       switch (direction) {
315       case 'fromString':
316         return 'oops';
317       case 'toString':
318         var d  = zeroPad(2, value.getUTCDate()),
319             m  = zeroPad(2, value.getUTCMonth() + 1),
320             y  = zeroPad(4, value.getUTCFullYear()),
321             hh = zeroPad(2, value.getUTCHours()),
322             mm = zeroPad(2, value.getUTCMinutes()),
323             ss = zeroPad(2, value.getUTCSeconds()),
324             ms = zeroPad(4, value.getUTCMilliseconds());
325         return y + '-' + m + '-' + d + 'T' + hh + ':' + mm + ':' + ss + '.' + ms + '000Z';
326       default:
327         return undefined;
328       }
329     }
330   };

332   function objectToSubscriptionOptionsForm(options) {
333     var xData = OX.XML.XMPP.XDataForm.create({type: 'submit'}),
334         opts  = OX.XML.Element.extend({name: 'options'}).create({}, xData);
335 
336     xData.addField('FORM_TYPE', 'http://jabber.org/protocol/pubsub#subscribe_options');
337 
338     for (var o in options) {
339       if (options.hasOwnProperty(o)) {
340         var trVal = options[o];
341         if (optionTransforms[o]) {
342           trVal = optionTransforms[o]('toString', trVal);
343         }
344         xData.addField('pubsub#' + o, trVal);
345       }
346     }
347 
348     return opts;
349   }

351   doConfigureNodeSubscription = function (subscription, subOptions, callbacks, options) {
352     var iq = OX.XML.XMPP.IQ.extend(),
353         pubsub = OX.XML.XMPP.PubSub.extend();
354 
355     iq.to(this.pubSubURI.path);
356     iq.type('set');
357     iq.addChild(pubsub);
358 
359     var opts = objectToSubscriptionOptionsForm.call(this, subOptions);
360     opts.attr('node', subscription.node);
361     opts.attr('jid', subscription.jid);
362     opts.attr('subid', subscription.subid);
363 
364     pubsub.addChild(opts);
365 
366     var that = this;
367     var wrappedCb = function () {
368       configureNodeSubscriptionHandler.apply(that, arguments);
369     };
370 
371     this.connection.send(iq.toString(), wrappedCb, [subscription, subOptions, callbacks, options]);
372   };

374   doSubscribe = function (node, subOptions, callbacks, options) {
375     var iq        = OX.XML.XMPP.IQ.extend(),
376         pubsub    = OX.XML.XMPP.PubSub.extend(),
377         subscribe = OX.XML.Element.extend({name: 'subscribe'});
378 
379     iq.to(this.pubSubURI.path);
380     iq.type('set');
381     subscribe.attr('node', node);
382     subscribe.attr('jid', this.connection.getJID());
383     pubsub.addChild(subscribe);
384 
385     if (subOptions) {
386       var opts = objectToSubscriptionOptionsForm.call(this, subOptions);
387       pubsub.addChild(opts);
388     }
389     iq.addChild(pubsub);
390 
391     var that = this;
392     var cb = function () {
393       subscriptionHandler.apply(that, arguments);
394     };
395 
396     this.connection.send(iq.toString(), cb, [node, subOptions, callbacks, options]);
397   };

399   doGetSubscriptions = function (node, callbacks, options) {
400     var iq = OX.XML.XMPP.IQ.extend(),
401         pub = OX.XML.XMPP.PubSub.extend(),
402         sub = OX.XML.Element.extend({name: 'subscriptions'});
403 
404     iq.to(this.pubSubURI.path);
405     iq.type('get');
406 
407     if (node) {
408       sub.attr('node', node);
409     }
410 
411     pub.addChild(sub);
412     iq.addChild(pub);
413 
414     var that = this;
415     var wrappedCb = function () {
416       getSubscriptionsHandler.apply(that, arguments);
417     };
418 
419     this.connection.send(iq.toString(), wrappedCb, [node, callbacks, options]);

484     /**
485      * Get subscriptions on a node.
486      *
487      * Passing an initial <tt>node</tt> parameter retrieves subscriptions on the requested
488      * node.  Otherwise a single parameter of <tt>callbacks</tt> requests all subscriptions
489      * at all nodes of the pubsub service.
490      *
491      * @see <a href="http://xmpp.org/extensions/xep-0060.html#entity-subscriptions">XEP: 0060 - Entity Subscriptions</a>
492      *
493      * @param {Object} callbacks an object supplying functions for 'onSuccess' and 'onError'
494      *   @param {Function} callbacks.onSuccess The success callback.
495      *     @param {OX.URI} callbacks.onSuccess.requestedURI The URI you requested.
496      *     @param {OX.URI} callbacks.onSuccess.finalURI The redirected URI that your requested URI maps to.
497      *     @param {OX.Subscription[]} callbacks.onSuccess.subscriptions The subscriptions associated with the finalURI.
498      *     @param {OX.PacketAdapter} callbacks.onSuccess.packet The packet recieved.
499      *   @param {Function} callbacks.onError The error callback.
500      *     @param {OX.URI} callbacks.onError.requestedURI The URI you requested.
501      *     @param {OX.URI} callbacks.onError.finalURI The redirected URI that your requested URI maps to.
502      *     @param {OX.PacketAdapter} callbacks.onError.packet The packet recieved.
503      * @param {Object} [options]
504      *   @param {String} options.node The node name to request subscriptions on. Omitting the node name implies all nodes
505      *   @param {Boolean} options.strict Only apply callbacks to subscriptions that match the exact JID as the current connection.
506      * @returns {void}
507      * This will NOT match a bare JID to a full JID.
508      *
509      * @example
510      *   service.getSubscriptions({
511      *     onSuccess: function (requestedURI, finalURI, subscriptions, packet) {},
512      *     onError: function (requestedURI, finalURI, packet)
513      *   }, {node: '/'});
514      *
515      * @example
516      *   service.getSubscriptions({
517      *     onSuccess: function (requestedURI, finalURI, subscriptions, packet) {},
518      *     onError: function (requestedURI, finalURI, packet)
519      *   });
520      */

548     /**
549      * Subscribe to a node.
550      *
551      * @param {String} node The node ID to subscribe to.
552      * @param {Object} [subOptions] Subscription options
553      * @param {Object} [callbacks] an object supplying functions for 'onSuccess', and 'onError'.
554      *   @param {Function} [callbacks.onSuccess] The success callback.
555      *     @param {OX.URI} [callbacks.onSuccess.requestedURI] The URI you requested.
556      *     @param {OX.URI} [callbacks.onSuccess.finalURI] The redirected URI that your requested URI maps to.
557      *     @param {OX.PacketAdapter} [callbacks.onSuccess.packet] The packet of the returned subscription.
558      *   @param {Function} [callbacks.onError] The error callback.
559      *     @param {OX.URI} [callbacks.onError.requestedURI] The URI you requested.
560      *     @param {OX.URI} [callbacks.onError.finalURI] The redirected URI that your requested URI maps to.
561      *     @param {OX.PacketAdapter} [callbacks.onSuccess.packet] The packet that contains the error.
562      * @param {Object} [options] Optional arguments.
563      *   @param {Boolean} [options.reuseSubscriptions] Tells subscribe to lookup subscriptions on the requested node, then reuse any if they exist. If none exist, it will create a new subscription.
564      * @returns {void}
565      *
566      * @example
567      *   var subOptions = {expire: new Date()};
568      *   service.subscribe('/', subOptions, {
569      *     onSuccess: function (requestedURI, finalURI) {},
570      *     onError:   function (requestedURI, finalURI) {}
571      *   });
572      */
573     subscribe: function (node, subOptions, callbacks, options) {
574       options = options || {};
575       callbacks = callbacks || {};
576 
577       var that = this;
578       if (options.reuseSubscriptions) {
579         this.getSubscriptions({
580           onSuccess: function (requestedURI, finalURI, subs, packet) {
581             var resubscribed = false,
582                 finalNode = finalURI.queryParam('node');
583 
584             for (var i = 0, len = subs.length; i < len; i++) {
585               if (subs[i].subscription === 'subscribed') {
586                 subs[i].node = finalURI.queryParam('node'); // Node is not set when specifying a node.
587                 that.configureNodeSubscription(subs[i], subOptions, callbacks, {
588                   origURI: requestedURI
589                 });
590                 resubscribed = true;
591                 break;
592               }
593             }
594 
595             if (!resubscribed) {
596               options.reuseSubscriptions = false;
597               that.subscribe(node, subOptions, callbacks, options);
598             }
599           },
600           onError: callbacks.onError
601         }, { strict: true, node: node });
602       } else {
603         doSubscribe.call(this, node, subOptions, callbacks, options);
604       }
605     },
