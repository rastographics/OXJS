1 /**
  2  * @namespace
  3  * CallDialog mixin.
  4  *
  5  * @requires A 'connectionAdapter' property which is an {@link OX.ConnectionAdapter} object on receiving object.
  6  * @requires A 'callID' property on receiving object.
  7  * @requires A 'fromTag' property on receiving object.
  8  * @requires A 'toTag' property on receiving object.
  9  */
 10 OX.Mixin.CallDialog = /** @lends OX.Mixin.CallDialog# */{
 11   /**
 12    * Transfer a call to a sip address.
 13    *
 14    * @param {String} targetURI To what SIP URI to transfer the active call.
 15    * @param {String} endpoint Either 'caller' or 'callee'
 16    * @param {Object} [callbacks] An object supplying functions for 'onSuccess', and 'onError'.
 17    *   @param {Function} [callbacks.onSuccess] The success callback.
 18    *   @param {Function} [callbacks.onError] The error callback.
 19    *     @param {OX.PacketAdapter} [callbacks.onError.packet] The packet recieved.
 20    * @returns {void}
 21    *
 22    * @see <a href="http://wiki.onsip.com/docs/Active-Calls_Component#transfer">Transferring a call</a>
 23    * @example
 24    *   call.transfer('sip:lisa@example.com', 'callee');
 25    */
 26   transfer: function (targetURI, endpoint, callbacks) {
 27     var iq    = OX.XML.XMPP.IQ.extend(),
 28         cmd   = OX.XML.XMPP.Command.extend(),
 29         xData = OX.XML.XMPP.XDataForm.extend(),
 30         uri   = OX.Settings.URIs.command.transferCall;
 31 
 32     callbacks = callbacks || {};
 33 
 34     iq.to(uri.path);
 35     iq.type('set');
 36     cmd.node(uri.queryParam('node'));
 37     xData.type('submit');
 38     xData.addField('call-id',    this.callID);
 39     xData.addField('from-tag',   this.fromTag);
 40     xData.addField('to-tag',     this.toTag);
 41     xData.addField('target-uri', targetURI);
 42     xData.addField('endpoint',   endpoint);
 43 
 44     iq.addChild(cmd.addChild(xData));
 45 
 46     this.connection.send(iq.toString(), function (packet) {
 47       if (!packet) {
 48         return;
 49       }
 50 
 51       if (packet.getType() === 'error' && callbacks.onError) {
 52         callbacks.onError(packet);
 53       } else if (callbacks.onSuccess) {
 54         callbacks.onSuccess();
 55       }
 56     }, []);
 57   },
 58
