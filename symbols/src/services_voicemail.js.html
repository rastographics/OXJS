1 /**
  2  * @namespace
  3  * <p>Namespace for voicemail related services.</p>
  4  * <p>The voicemail pubsub service allows subscribed users to
  5  * receive updates regarding the status of their voicemail account(s),
  6  * e.g. added, deleted, or moved voicemail messages.</p>
  7  *
  8  * @extends OX.Base
  9  * @extends OX.Mixin.Subscribable
 10  * @requires A 'connectionAdapter' property inherited from an {@link OX.Connection}.
 11  * @see <a href="http://wiki.onsip.com/docs/Voicemail_Pubsub">Voicemail PubSub</a>
 12  */
 13 OX.Service.Voicemail = OX.Base.extend(OX.Mixin.Subscribable, function () {
 14   function itemType(element) {
 15     if (!element) {
 16       return undefined;
 17     } else if (element.getElementsByTagName('voicemail').length > 0) {
 18       return 'voicemail';
 19     } else if (element.getElementsByTagName('labels').length > 0) {
 20       return 'labels';
 21     } else {
 22       return undefined;
 23     }
 24   }
 25 
 26   function voicemailItem(element) {
 27     if (!element) {
 28       return undefined;
 29     }
 30 
 31     var rc = {};
 32     var voicemailNode = element.getElementsByTagName('voicemail');
 33 
 34     if (!voicemailNode || !voicemailNode[0]) {
 35       return undefined;
 36     }
 37 
 38     var children = voicemailNode[0].childNodes;
 39     for (var i = 0, len = children.length; i < len; i++) {
 40       var node = children[i];
 41 
 42       if (!node.nodeName || !node.firstChild) {
 43         continue;
 44       }
 45 
 46       switch (node.nodeName.toLowerCase()) {
 47       case 'mailbox':
 48         rc.mailbox = parseInt(node.firstChild.nodeValue, 10);
 49         break;
 50       case 'caller-id':
 51         rc.callerID = node.firstChild.nodeValue;
 52         break;
 53       case 'created':
 54         rc.created = node.firstChild.nodeValue;
 55         break;
 56       case 'sipfrom':
 57         rc.sipfrom = node.firstChild.nodeValue;
 58         break;
 59       case 'duration':
 60         rc.duration = parseInt(node.firstChild.nodeValue, 10);
 61         break;
 62       case 'labels':
 63         var labels = [];
 64         for (var j = 0, jlen = node.childNodes.length; j < jlen; j++) {
 65           var elt = node.childNodes[j];
 66           if (elt.tagName && elt.tagName === 'label') {
 67             labels.push(elt.firstChild.nodeValue);
 68           }
 69         }
 70         rc.labels = labels;
 71         break;
 72       }
 73     }
 74     return rc;
 75   }
 76 
 77   function labelItem(element) {
 78     if (!element) {
 79       return undefined;
 80     }
 81 
 82     var rc = {labels: []};
 83     var labelsNode = element.getElementsByTagName('labels');
 84 
 85     if (!labelsNode || !labelsNode[0]) {
 86       return undefined;
 87     }
 88 
 89     var children = labelsNode[0].childNodes;
 90     for (var i = 0, len = children.length; i < len; i++) {
 91       var node = children[i];
 92 
 93       if (node.nodeName && node.nodeName === 'label') {
 94         rc.labels.push(node.firstChild.nodeValue);
 95       }
 96     }
 97     return rc;
 98   }
 99 
100   return /** @lends OX.Service.Voicemail */{
101     /**
102      * URI for this PubSub service.
103      */
104     pubSubURI: OX.Settings.URIs.pubSub.voicemail,
105 
106     /**
107      * @name OX.Service.Voicemail.Item
108      * @namespace
109      * Voicemail Item.
110      * @extends OX.Item
111      */
112     Item: OX.Item.extend(/** @lends OX.Service.Voicemail.Item# */{
113       /** The mailbox number for this voicemail. */
114       mailbox:  null,
115 
116       /** The caller ID of this voicemail. */
117       callerID: null,
118 
119       /** The SIP address this voicemail is from. */
120       sipfrom: null,
121 
122       /** The time this voicemail was created. */
123       created:  null,
124 
125       /** How long, in seconds, this voicemail is. */
126       duration: null,
127 
128       /** An array of labels for this voicemail. */
129       labels:   null,
130 
131       /**
132        * Cache this Voicemail.
133        *
134        * @param {Object} [callbacks] Callbacks with 'onSuccess' and 'onError'
135        *   @param {Function} [callbacks.onSuccess] The success callback
136        *     @param {OX.PacketAdapter} [callbacks.onSuccess.packet] The packet recieved.
137        *   @param {Function} [callbacks.onError] The error callback
138        *     @param {OX.PacketAdapter} [callbacks.onError.packet] The packet recieved.
139        * @returns {void}
140        *
141        * @see <a href="http://wiki.onsip.com/docs/Voicemail_Component#cache">Voicemail caching</a>
142        */
143       cacheMessage: function (callbacks) {
144         var iq    = OX.XML.XMPP.IQ.extend(),
145             cmd   = OX.XML.XMPP.Command.extend(),
146             xData = OX.XML.XMPP.XDataForm.extend(),
147             uri   = OX.Settings.URIs.command.cacheVoicemail,
148             node_parts     = this.uri.queryParam('node').split('/'),
149             vm_sip_address = node_parts[2] + '@' + node_parts[1],
150             vm_id          = this.uri.queryParam('item');
151 
152         callbacks = callbacks || {};
153 
154         iq.to(uri.path);
155         iq.type('set');
156         cmd.node(uri.queryParam('node'));
157         xData.type('submit');
158         xData.addField('vm-sip-address', vm_sip_address);
159         xData.addField('vm-id', vm_id);
160 
161         iq.addChild(cmd.addChild(xData));
162 
163         this.connection.send(iq.toString(), function (packet) {
164           if (!packet) {
165             return;
166           }
167           if (packet.getType() === 'error' && callbacks.onError &&
168               callbacks.onError.constructor === Function) {
169             callbacks.onError(packet);
170           } else if (callbacks.onSuccess && callbacks.onSuccess.constructor === Function) {
171             callbacks.onSuccess(packet);
172           }
173         }, []);
174       },
175 
176       /**
177        * Delete this Voicemail.
178        *
179        * @param {Object} [callbacks] Callbacks with 'onSuccess' and 'onError'
180        *   @param {Function} [callbacks.onSuccess] The success callback
181        *     @param {OX.PacketAdapter} [callbacks.onSuccess.packet] The packet recieved.
182        *   @param {Function} [callbacks.onError] The error callback
183        *     @param {OX.PacketAdapter} [callbacks.onError.packet] The packet recieved.
184        * @returns {void}
185        *
186        * @see <a href="http://wiki.onsip.com/docs/Voicemail_Component#delete">Deleting voicemail</a>
187        */

193             node_parts     = this.uri.queryParam('node').split('/'),
194             vm_sip_address = node_parts[2] + '@' + node_parts[1],
195             vm_id          = this.uri.queryParam('item');

216             callbacks.onSuccess(packet);

218         }, []);
219       }

225      * @namespace

228      * @extends OX.Item
229      */
230     LabelItem: OX.Item.extend(/** @lends OX.Service.Voicemail.LabelItem#*/{
231       /** An array of all voicemail labels. */
232       labels: null

235     /** @private
236      * Returns an {@link OX.Service.Voicemail.Item} or {@link OX.Service.Voicemail.LabelItem}
237      * from an XML Document.
238      * This method should be called once for each item to be constructed.
239      * If a DOMElement contains more than one item node, only the first
240      * item node will be returned as an OX.Service.Voicemail.Item
241      *
242      * @param {Element|Node} element The DOM Element or Node to parse into a {@link OX.Service.Voicemail.Item} or
243      *                               {@link OX.Service.Voicemail.LabelItem}.
244      * @returns {OX.Service.Voicemail.Item} The item created from the element passed in.
245      */
246     itemFromElement: function (element) {
247       var rc, item;
248 
249       switch (itemType(element)) {
250       case 'voicemail':
251         item = voicemailItem(element);
252         if (item) {
253           rc = this.Item.extend(item, {connection: this.connection});
254         }
255         break;
256       case 'labels':
257         item = labelItem(element);
258         if (item) {
259           rc = this.LabelItem.extend(item, {connection: this.connection});
260         }
261         break;
262       }
263 
264       return rc;
265     }
266   };
